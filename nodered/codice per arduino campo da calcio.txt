#include <SPI.h>
#include <Ethernet.h>
#include <PubSubClient.h>
#include <DHT.h>
#include <EmonLib.h>


// Update these with values suitable for your network.
byte mac[]    = {  0xDE, 0xED, 0xBA, 0xFE, 0xFE, 0xED };
//IPAddress ip(172, 16, 0, 100);
IPAddress server(192, 168, 1, 13);//--------->inserire qui l'ip del raspberry
EthernetClient ethClient;
PubSubClient client(ethClient);

EnergyMonitor emon1;//monitoraggio corrente
EnergyMonitor emon2;
EnergyMonitor emon3;

#define RELAY1 6 // pin digitali ai quali sono connessi i relè
#define RELAY2 7
#define RELAY3 8
#define RELAY4 9
#define RELAY5 10
#define RELAY6 15
#define RELAY7 16
#define RELAY8 17

void setup()
{
  Serial.begin(9600);
  
  Ethernet.begin(mac);  //setup parte internet
  client.setServer(server, 1884);
  client.setCallback(callback);
  
  pinMode(RELAY1, OUTPUT); //setto la modalità dei pin digitali relativi ai relè
  pinMode(RELAY2, OUTPUT);
  pinMode(RELAY3, OUTPUT);
  pinMode(RELAY4, OUTPUT);
  pinMode(RELAY5, OUTPUT);
  pinMode(RELAY6, OUTPUT);
  pinMode(RELAY7, OUTPUT);
  pinMode(RELAY8, OUTPUT);
  
  emon1.current(1,30);  // Voltage: input pin, calibration, phase_shift
  emon2.current(2,30);
  emon3.current(3,30);
  
  delay(1500);
}

void loop()
{
  if (!client.connected()) {
      Serial.print("Attempting MQTT connection...");
      // Attempt to connect
      if (client.connect("arduinoClient")) {
        Serial.println("connected");
        // Once connected, publish an announcement...
         client.subscribe("RichiediUmidita",0);
         client.subscribe("MisuraConsumo",0);
         client.subscribe("AttivaRele",0);
         client.subscribe("DisattivaRele",0);
        
        // ... and resubscribe
       
      } else {
        Serial.print("failed, rc=");
        Serial.print(client.state());
        Serial.println(" try again in 5 seconds");
        // Wait 5 seconds before retrying
       delay(5000);
      }
    }
  client.loop();
}





void callback(char* topic, byte* payload, unsigned int length) {

  Serial.print("] ");
  String message = "";
  String top = String(topic);
  for (int i=0;i<length;i++) {
    Serial.print((char)payload[i]);
    message += (char)payload[i];
  }
  Serial.println();

 }
 
 if(top == "DisattivaRele"){

  strncpy(val, disattivo.c_str(), m);

 if(top == "MisuraConsumo1"){

String AttivaRele(int numerorele) {
  if (numerorele == 1)
    digitalWrite(RELAY1, HIGH);
  if (numerorele == 2)
    digitalWrite(RELAY2, HIGH);
  if (numerorele == 3)
    digitalWrite(RELAY3, HIGH);
  if (numerorele == 4)
    digitalWrite(RELAY4, HIGH);
  if (numerorele == 5)
    digitalWrite(RELAY5, HIGH);
  if (numerorele == 6)
    digitalWrite(RELAY6, HIGH);
  if (numerorele == 7)
    digitalWrite(RELAY7, HIGH);
  if (numerorele == 8)
    digitalWrite(RELAY8, HIGH);
  String ret = "on";
  return ret;
}
String DisattivaRele(int numerorele) {
  if (numerorele == 1)
    digitalWrite(RELAY1, LOW);
  if (numerorele == 2)
    digitalWrite(RELAY2, LOW);
  if (numerorele == 3)
    digitalWrite(RELAY3, LOW);
  if (numerorele == 4)
    digitalWrite(RELAY4, LOW);
  if (numerorele == 5)
    digitalWrite(RELAY5, LOW);
  if (numerorele == 6)
    digitalWrite(RELAY6, LOW);
  if (numerorele == 7)
    digitalWrite(RELAY7, LOW);
  if (numerorele == 8)
    digitalWrite(RELAY8, LOW);
  String ret = "off";
  return ret;
}
String MisuraConsumo1() {

  Serial.println(Irms*230.0);//Scrivo sul monitor seriale Corrente*Tensione=Potenza

}
